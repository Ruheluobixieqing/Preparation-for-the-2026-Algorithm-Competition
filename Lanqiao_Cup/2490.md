# 2490.小蓝的括号串 1

用数组模拟栈

实际上，本地并不需要真的将 '(' 或 ')' 存入数组，只需要判断是否合法即可

```cpp
#include<bits/stdc++.h>
const int N = 2e5+3;
int stk[N], top = 0;

int main()
{
    std::ios::sync_with_stdio(false),std::cin.tie(nullptr),std::cout.tie(nullptr);
    int n;
    std::string s;
    std::cin>>n>>s;
    if (n&1)
    {
        std::cout<<"No"<<'\n';
        return 0;
    }
    int len = s.size();
    for (int i = 0; i < len; i++)
    {
        if (s[i] == '(') top++;
        else
        {
            if (top) top--;
            else
            {
                std::cout<<"No"<<'\n';
                return 0;
            }
        }
    }
    if (top) std::cout<<"No"<<'\n';
    else std::cout<<"Yes"<<'\n';
    return 0;
}
```
