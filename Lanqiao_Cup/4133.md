# 4133.小浩的 ABC

```cpp
#include <bits/stdc++.h>
#define ll long long

void solve()
{
    ll x;
    std::cin >> x;
    if (x == 1)
    {
        // 该情况无解
        std::cout<<"-1"<<'\n';
        return;
    }
    else if ( x <= 1e6 )
    {
        std::cout<<x-1<<" 1 1"<<'\n';
        return;
    }
    else
    {
        ll a = 1e6;
        ll c = x % a;
        if (c == 0) c = 1e6;
        ll b = (x - c) / a;
        std::cout<<a<<' '<<b<<' '<<c<<'\n';
    }
    return;
}


int main()
{
    std::ios::sync_with_stdio(false),std::cin.tie(nullptr),std::cout.tie(nullptr);
    int t;
    std::cin>>t;
    while (t--) solve();
    return 0;
}
```
