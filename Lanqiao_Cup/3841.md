# 3841.单身贵族游戏

- 本题我的直观感受是直接拟棋盘结构
- 也可以直接处理读入数据，不过需要较为复杂的判断相邻处理，不如直接模拟棋盘来的简单
- 对于那些缺损的结构，可以用其他符号补全，只要不影响后续的判断即可

```cpp
#include <bits/stdc++.h>
const int N = 10;
int arr[N][N];

bool check(int i,int j)
{
    if (arr[i+1][j] ==1 || arr[i-1][j] == 1 || arr[i][j+1] == 1 || arr[i][j-1] == 1) return true;
    else return false;
}

int main()
{
    std::ios::sync_with_stdio(false),std::cin.tie(nullptr),std::cout.tie(nullptr);
    for (int i = 1; i <= 7; i++)
    {
        std::string s;
        std::cin>>s;
        int p = 0;
        for (int j = 1; j <= 7; j++)
        {
            if ((i < 3 || i > 5) && (j < 3 || j > 5))
            {
                arr[i][j] = 9;
                continue;
            }
            arr[i][j] = s[p++]-'0';
        }
    }
    bool ans = false;
    for (int i = 1; i <= 7; i++) for (int j = 1; j <= 7; j++) if (arr[i][j] == 1 && check(i,j)) ans = true;
    if (ans) std::cout<<"no"<<'\n';
    else std::cout<<"yes"<<'\n';
    return 0;
}
```
