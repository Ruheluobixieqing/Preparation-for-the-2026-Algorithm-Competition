# 549.扫雷

- 模拟即可，注意数组越界问题，难度不大

```cpp
#include<bits/stdc++.h>
const int N = 1e2+3;
int arr[N][N];

int search(int x, int y)
{
    int ans = 0;
    for (int i = x -1; i <= x + 1; i++) for (int j = y - 1; j <= y + 1; j++) if (arr[i][j] == 9) ans++;
    return ans;
}

int main()
{
    std::ios::sync_with_stdio(false),std::cin.tie(nullptr),std::cout.tie(nullptr);
    int n,m;
    std::cin>>n>>m;
    for(int i = 1;i <= n; i++)
    {
        for (int j = 1; j <= m; j++)
        {
            int num = -1;
            std::cin>>num;
            if (num == 1) num = 9;
            arr[i][j] = num;
        }
    }
    for (int i = 1; i <= n; i++)
    {
        for (int j = 1;j <= m; j++)
        {
            if (arr[i][j] != 9 ) arr[i][j] = search(i,j);
            std::cout<<arr[i][j]<<" \n"[j == m];
        }
    }
    return 0;
}
```
