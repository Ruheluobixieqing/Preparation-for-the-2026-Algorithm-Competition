# 5379.第 k 大的数

脑子宕机了，这题其实就是维护一个小根堆就可以了，不需要考虑复杂的其他关系

注意一下如何将默认的大根堆改为小根堆就可以了

```cpp
#include<bits/stdc++.h>
#define ll long long
std::priority_queue<int, std::vector<int>, std::greater<int>> pq;
int main()
{
    std::ios::sync_with_stdio(false),std::cin.tie(nullptr),std::cout.tie(nullptr);
    int n, k;
    std::cin >> n >> k;
    for (int i = 0; i < n; i++)
    {
        int x;
        std::cin >> x;
        pq.push(x);
    }

    for (int i = 0; i < k; i++)
    {
        int x;
        std::cin >> x;
        if (x > pq.top())
        {
            pq.pop();
            pq.push(x);
        }
        std::cout << pq.top() << ' ';
    }
    return 0;
}
```
