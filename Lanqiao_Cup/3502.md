# 3502.平方差

这题的 O(1) 解法比较考验思维能力

```cpp
#include<bits/stdc++.h>
#define ll long long

ll f(ll n)
{
    ll group = n / 4;
    ll x = n % 4;
    if (x == 2) return group * 3 + 2;
    else return group * 3 + x + 1;
}

int main()
{
    std::ios::sync_with_stdio(false),std::cin.tie(nullptr),std::cout.tie(nullptr);
    ll l,r;
    std::cin>>l>>r;
    ll ans = f(r) - f(l-1);
    std::cout<<ans<<'\n';
    return 0;
}
```
