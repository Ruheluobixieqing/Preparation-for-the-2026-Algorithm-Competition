# 168.倍数问题

这题梳理下来其实不算太难，但是分类讨论还是比较复杂的

需要重点关注的部分是 i j t 三个数在不同大小关系下的不同计算

```cpp
#include <bits/stdc++.h>
const int N = 1e5+3;
int a[N];
std::vector<int> v[N];

int main()
{
    std::ios::sync_with_stdio(false),std::cin.tie(nullptr),std::cout.tie(nullptr);
    int n, K;
    std::cin>>n>>K;
    for (int i = 1; i <= n; i++) std::cin>>a[i];
    for (int i = 1; i <= n; i++) v[a[i] % K].push_back(a[i]);
    for (int i = 0; i < K; i++) std::sort(v[i].begin(), v[i].end(), [](int u, int v)
    {
        return u > v;
    });
    int ans = 0;

    for (int i = 0; i < K; i++)
    {
        if (!v[i].size()) continue;
        for (int j = i; j < K; j++)
        {
            int t = (2 * K - i - j ) % K;
            if (t < j || !v[j].size() || !v[t].size()) continue;
            if (i == j == t)
            {
                if (v[i].size() < 3) continue;
                ans = std::max(ans, v[i][0] + v[i][1] + v[i][2]);
            }
            else if (i == j && j < t)
            {
                if (v[i].size() < 2) continue;
                ans = std::max(ans, v[i][0] + v[i][1] + v[t][0]);
            }
            else if (i < j && j == t)
            {
                if (v[j].size() < 2) continue;
                ans = std::max(ans, v[i][0] + v[j][0] + v[j][1]);
            }
            else if (i < j && j < t)
            {
                ans = std::max(ans, v[i][0] + v[j][0] + v[t][0]);
            }
        }
    }

    std::cout<<ans<<'\n';
    return 0;
}
```
