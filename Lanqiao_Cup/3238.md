# 3238.小蓝和小桥的挑战

这题比较简单，直接上代码

```cpp
#include<bits/stdc++.h>

void solve()
{
    int n;
    std::cin>>n;
    int sum = 0, ans = 0;
    for (int i = 0; i < n; i++)
    {
        int x;
        std::cin>>x;
        sum += x;
        if (!x) ans++;
    }
    sum += ans;
    if (!sum) ans++;
    std::cout<<ans<<'\n';
}

int main()
{
    std::ios::sync_with_stdio(false),std::cin.tie(nullptr),std::cout.tie(nullptr);
    int t;
    std::cin>>t;
    while (t--) solve();
    return 0;
}
```
