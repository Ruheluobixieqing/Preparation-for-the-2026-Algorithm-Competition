# 192.反倍数

- 暴力枚举解空间

```cpp
#include <bits/stdc++.h>
int a,b,c;

bool judgeNum(int x)
{
  if(x % a ==0 || x % b == 0 || x % c == 0) return false;
  return true; 
}

int main()
{
  int n;
  std::cin>>n>>a>>b>>c;
  int ans = 0;
  for(int i = 1;i <=n;i++) if(judgeNum(i)) ans++;
  std::cout<<ans<<'\n';
  return 0;
}
```
