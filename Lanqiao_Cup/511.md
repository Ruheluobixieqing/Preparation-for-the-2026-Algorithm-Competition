# 511.机器翻译

用数组模拟队列

```cpp
#include<bits/stdc++.h>
const int N = 2e5+3;
int q[N];
std::map<int, int>mp;

int main()
{
    std::ios::sync_with_stdio(false),std::cin.tie(nullptr),std::cout.tie(nullptr);
    int n, m;
    std::cin>>m>>n;
    int head = 0,tail = -1;
    int ans = 0;
    for (int i = 0; i < n; i++)
    {
        int k;
        std::cin>>k;
        if (mp[k]) continue;
        mp[k]++, ans++;
        if (tail - head + 1 < m) q[++tail] = k;
        else
        {
            mp[q[head++]]--;
            q[++tail] = k;
        }
    }
    std::cout<<ans<<'\n';
    return 0;
}
```
