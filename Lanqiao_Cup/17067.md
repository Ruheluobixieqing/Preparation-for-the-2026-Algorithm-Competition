# 17067.小新的质数序列挑战

```cpp
#include <bits/stdc++.h>
#define ll long long

void solve()
{
    ll a, b;
    std::cin>>a>>b;
    if (a == 1 || b == 1)
    {
        std::cout<<"-1"<<'\n';
        return;
    }
    else
    {
        int ans = std::__gcd(a, b) == 1 ? 1 : 0;
        std::cout<<ans<<'\n';
        return;
    }
}

int main()
{
    std::ios::sync_with_stdio(false),std::cin.tie(nullptr),std::cout.tie(nullptr);
    int t;
    std::cin>>t;
    while (t--) solve();
    return 0;
}
```
