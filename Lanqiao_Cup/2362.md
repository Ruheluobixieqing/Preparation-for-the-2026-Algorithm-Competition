# 2362.Fibonacci 集合

这题可以用 set 容器解决，需要注意，一定要保证 set 元素溢出足够多，否则可能得到的数并不是真正的 2020 小

```cpp
#include<bits/stdc++.h>
#define ull unsigned long long
std::set<ull> st;

int main()
{
    std::ios::sync_with_stdio(false),std::cin.tie(nullptr),std::cout.tie(nullptr);
    st.insert(1),st.insert(2),st.insert(3);
    for (auto &e: st)
    {
        st.insert(3*e+2),st.insert(5*e+3),st.insert(8*e+5);
        if (st.size() > 2e5) break;
    }
    int time = 1;
    for (auto &e: st)
    {
        if (time != 2020)
        {
            time ++;
        }
        else
        {
            std::cout<< e << '\n';
            return 0;
        }
    }
}
```
