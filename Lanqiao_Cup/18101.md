# 18101.聚合一块

```cpp
#include<bits/stdc++.h>
const int N = 2e5+3;
int pre[N], size[N];

int getRoot(int x)
{
    return pre[x] == x ? x : getRoot(pre[x]);
}
void merge(int x, int y)
{
    int rx = getRoot(x), ry = getRoot(y);
    if (size[rx] > size[ry]) std::swap(rx, ry);

    pre[rx] = ry;
    size[ry] += size[rx];
}

int main()
{
    std::ios::sync_with_stdio(false),std::cin.tie(nullptr),std::cout.tie(nullptr);
    int n, m;
    std::cin >> n >> m;
    for (int i = 1; i <= n; i++) pre[i] = i, size[i] = 1;
    for (int i = 1; i <= m; i++)
    {
        int x, y;
        std::cin >> x >> y;
        merge(x, y);
    }
    int ans = 0;
    for (int i = 1; i <= n; i++) if (getRoot(i) == i) ans++;
    std::cout << ans - 1 << '\n';
    return 0;
}
```
