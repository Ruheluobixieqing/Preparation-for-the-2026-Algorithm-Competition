# 191.特别数的和

- 暴力枚举解空间即可

```cpp
#include <bits/stdc++.h>

bool checkNum(int x)
{
  while(x)
  {
    int k = x%10;
    if(k == 2 || k == 0 || k == 1 || k == 9) return true;
    x /= 10;
  }
  return false;
}

int main()
{
  int n;
  std::cin>>n;
  int ans = 0;
  for(int i = 1; i <= n; i++) if (checkNum(i)) ans += i;
  std::cout<<ans<<'\n';
  return 0;
}
```
