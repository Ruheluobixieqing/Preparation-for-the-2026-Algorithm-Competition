# JZ25.合并两个排序的链表

双指针，一遍比较一边遍历即可

注意特判 p1、p2 为空指针的情况

```cpp
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param pHead1 ListNode类 
     * @param pHead2 ListNode类 
     * @return ListNode类
     */
    ListNode* Merge(ListNode* pHead1, ListNode* pHead2) {
        // write code here
        ListNode *p1 = pHead1, *p2 = pHead2;
        ListNode *pHead;
        if(p1 == nullptr && p2 == nullptr) return nullptr;
        if(p1 == nullptr) return p2;
        else if(p2 == nullptr) return p1;
        else
        {
            if(p1 -> val <= p2 -> val)
            {
                pHead = p1;
                p1 = p1 -> next;
            }
            else
            {
                pHead = p2;
                p2 = p2 -> next;
            }
        }
        ListNode *p = pHead;
        while( p1 != nullptr || p2 != nullptr)
        {
            if(p1 == nullptr)
            {
                p -> next = p2;
                p2 = p2 -> next;
                p = p -> next;
            }
            else if(p2 == nullptr)
            {
                p -> next = p1;
                p1 = p1 -> next;
                p = p -> next;
            }
            else
            {
                if(p1 -> val <= p2 -> val)
                {
                    p -> next = p1;
                    p1 = p1 -> next;
                    p = p -> next;
                }
                else 
                {
                    p -> next = p2;
                    p2 = p2 -> next;
                    p = p -> next;
                }
            }
        }
        return pHead;
    }
};
```
