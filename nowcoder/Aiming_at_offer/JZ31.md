# JZ31.栈的压入、弹出序列

```cpp
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param pushV int整型vector 
     * @param popV int整型vector 
     * @return bool布尔型
     */
    std::stack<int> st;

    bool IsPopOrder(vector<int>& pushV, vector<int>& popV) {
        int n1 = pushV.size(), n2 = popV.size();
        int i = 0, j = 0;
        while(j < n2)
        {
            if(st.empty()) st.push(pushV[i++]);
            while(i < n1 && st.top() != popV[j]) st.push(pushV[i++]);
            if(st.top() != popV[j]) return false;
            // std::cout<< st.top() << ' ' << popV[j] << '\n';
            st.pop();
            j++;
        }
        return true;
    }
};
```
