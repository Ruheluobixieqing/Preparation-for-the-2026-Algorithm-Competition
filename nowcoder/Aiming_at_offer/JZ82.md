# JZ82.二叉树中和为某一值的路径（一）

DFS 即可

```cpp
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param root TreeNode类 
     * @param sum int整型 
     * @return bool布尔型
     */
    bool find = false;
    int tarSum = 0;
    void traverseTree(TreeNode* root, int nowSum)
    {
        nowSum += root -> val;
        if(nowSum == tarSum && root -> left == nullptr && root -> right == nullptr) find = true;
        if(root -> left) traverseTree(root -> left, nowSum);
        if(root -> right) traverseTree(root -> right, nowSum);
    }
    bool hasPathSum(TreeNode* root, int sum) {
        tarSum = sum;
        if(root) traverseTree(root, 0);
        return find;
    }
};
```
