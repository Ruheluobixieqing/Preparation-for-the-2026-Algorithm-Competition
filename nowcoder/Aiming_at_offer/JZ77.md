# JZ77.按之字形顺序打印二叉树

这个问题很有意思，之前一般都是按层遍历，这次不仅要按层遍历，还要区分遍历顺序是从左到右还是从右到左

用一个 nowRow 变量记录当前是奇数行还是偶数行，两种情况分开处理即可

```cpp
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param pRoot TreeNode类 
     * @return int整型vector<vector<>>
     */
    vector<vector<int> > Print(TreeNode* pRoot) {
        std::deque<TreeNode*> dq_odd,dq_even;
        std::vector<std::vector<int>> v;
        if(pRoot == nullptr) return v;
        dq_odd.push_back(pRoot);
        int nowRow = 1;
        std::vector<int> row;
        while(1)
        {
            if(nowRow&1)
            {
                // 当前为奇数行，从左到右遍历
                row.push_back(dq_odd.front() -> val);
                if(dq_odd.front() -> left != nullptr) dq_even.push_back(dq_odd.front() -> left);
                if(dq_odd.front() -> right != nullptr) dq_even.push_back(dq_odd.front() -> right);

                dq_odd.pop_front();
                if(dq_odd.empty())
                {
                    // 本行遍历完成
                    nowRow++;
                    v.push_back(row);
                    for(const auto &e: row) std::cout<< e << ' ';
                    std::cout<< '\n';
                    std::vector<int>().swap(row);
                    if(dq_even.empty()) return v;
                }
            }
            else
            {
                // 当前为偶数行，从右向左遍历
                row.push_back(dq_even.back() -> val);
                if(dq_even.back() -> right != nullptr) dq_odd.push_front(dq_even.back() -> right);
                if(dq_even.back() -> left != nullptr) dq_odd.push_front(dq_even.back() -> left);

                dq_even.pop_back();
                if(dq_even.empty())
                {
                    // 本行遍历完成
                    nowRow++;
                    v.push_back(row);
                    for(const auto &e: row) std::cout<< e << ' ';
                    std::cout<< '\n';
                    std::vector<int>().swap(row);
                    if(dq_odd.empty()) return v;
                }
            }
        }
    }
};
```
