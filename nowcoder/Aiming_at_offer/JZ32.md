# JZ32.从上到下打印二叉树

简单的树的层序遍历

```cpp
class Solution {
public:
    vector<int> PrintFromTopToBottom(TreeNode* root) {
		std::vector<int> ans;
		if(root == nullptr) return ans;
		// 层序遍历即可
		std::queue<TreeNode*> q;
		q.push(root);
		while(!q.empty())
		{
			if(q.front() -> left) q.push(q.front() -> left);
			if(q.front() -> right) q.push(q.front() -> right);
			ans.push_back(q.front() -> val);
			q.pop();
		}
		return ans;
    }
};
```
