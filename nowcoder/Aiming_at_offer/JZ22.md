# JZ22.链表中倒数最后 k 个结点

时间复杂度 O(2n)

```cpp
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param pHead ListNode类 
     * @param k int整型 
     * @return ListNode类
     */

    int getSize(ListNode* pHead){
        int n = 0;
        while(pHead)
        {
            pHead = pHead -> next;
            n ++;
        }
        return n;
    }
    ListNode* FindKthToTail(ListNode* pHead, int k) {
        // write code here
        int size = getSize(pHead);
        if(size < k) return nullptr;
        ListNode *p = pHead;
        for(int i = 1; i <= size - k; i++) p = p -> next;
        return p;
    }
};
```

也可以用栈进行存储

```cpp
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param pHead ListNode类 
     * @param k int整型 
     * @return ListNode类
     */
    ListNode* FindKthToTail(ListNode* pHead, int k) {
        // write code here
        std::stack<ListNode*> st;
        ListNode *p = pHead;
        while(p)
        {
            st.push(p);
            p =  p -> next;
        }
        st.push(nullptr);
        if(st.size() - 1 < k) return nullptr;
        else
        {
            while(k--) st.pop();
            return st.top();
        }
    }
};
```
