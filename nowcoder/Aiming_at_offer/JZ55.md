# JZ55.二叉树的深度

写一个递归函数计算树高即可，注意判断根结点是否为空，空树的高为 0

```cpp
class Solution {
public:
	int getDeepth(TreeNode* nowNode){
		if(nowNode -> left == nullptr && nowNode -> right == nullptr) return 1;
		if(nowNode -> left == nullptr)
		{
			return getDeepth(nowNode -> right) + 1;
		}
		else if(nowNode -> right == nullptr)
		{
			return getDeepth(nowNode -> left) + 1;
		}
		else return std::max(getDeepth(nowNode -> left), getDeepth(nowNode -> right)) + 1;
	}

    int TreeDepth(TreeNode* pRoot) {
		if(pRoot == nullptr) return 0;
		return getDeepth(pRoot);
    }
};
```
