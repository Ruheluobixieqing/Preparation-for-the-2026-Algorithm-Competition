# JZ30.包含 min 函数的栈

考虑到直接用 stack 没有什么诚意，这里且用 vector 手写一个简单栈

```cpp
class Solution {
private:
    // s1 为基础栈，s2 为 min 栈
    std::vector<int> s1, s2;
    int p1 = -1;
    int p2 = -1;

public:
    void push(int value) {
        p1++;
        s1.push_back(value);
        if(p2 == -1) s2.push_back(value);
        else {
            if(s2[p2] >= value) s2.push_back(value);
            else s2.push_back(s2[p2]);
        }
        p2++;
    }
    void pop() {
        p1--;
        p2--;
        s1.pop_back();
        s2.pop_back();
    }
    int top() {
        return s1[p1];
    }
    int min() {
        return s2[p2];
    }
};
```
