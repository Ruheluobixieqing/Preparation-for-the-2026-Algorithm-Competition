# JZ34.二叉树中和为某一值的路径（二）

和 JZ82 思路一致

```cpp
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param root TreeNode类 
     * @param target int整型 
     * @return int整型vector<vector<>>
     */
    int tarSum = 0;
    std::vector<std::vector<int>> ans;
    std::vector<int> v;
    void traverseTree(TreeNode* root, int nowSum)
    {
        v.push_back(root -> val);
        nowSum += root -> val;
        if(nowSum == tarSum && root -> left == nullptr && root -> right == nullptr) ans.push_back(v);
        if(root -> left) traverseTree(root -> left, nowSum);
        if(root -> right) traverseTree(root -> right, nowSum);
        v.pop_back();
    }
    vector<vector<int> > FindPath(TreeNode* root, int target) {
        tarSum = target;
        if(!root) return ans;
        traverseTree(root, 0);
        return ans;
    }
};
```
