# JZ11.旋转数组的最小数字

```cpp
class Solution {
public:
    int minNumberInRotateArray(vector<int>& nums) {
        int left = 0, right = nums.size() - 1;
        while(left < right){
            int mid = (left + right) >> 1;
            // left 到 mid 都是合法区间，说明答案在右侧
            if(nums[mid] > nums[right]) left = mid + 1;
            // 无法判断，只能逐个尝试
            else if(nums[mid] == nums[right]) right --;
            // left 到 mid 中有跳转
            else right = mid;
        }
        return nums[left];
    }
};
```
