# JZ73.翻转单词序列

用个栈存一下就好了

```cpp
class Solution {
public:

    std::stack<std::string> st;
    string ReverseSentence(string str) {
        if(str == "") return "";
        std::string s = "";
        for(auto e: str){
            if (e != ' ') {
                s += e;
                continue;
            } 
            else {
                if (s != "") {
                    st.push(s);
                    s = "";
                }
                continue;
            }
        }
        st.push(s);
        std::string ans = "";
        while(!st.empty()){
            ans = ans + st.top();
            st.pop();
            if(!st.empty()) ans += " ";
        }
        return ans;
    }
};
```
