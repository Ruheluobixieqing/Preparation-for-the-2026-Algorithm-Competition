# 11.盛水最多的容器

```cpp
// 时间复杂度不能是 O(n²)，故不考虑暴力


class Solution {
public:
    int maxArea(vector<int>& height) {
        int ans = 0;
        for(int left = 0, right = height.size() - 1; left < right;)
        {
            int sum = (right - left) * std::min(height[left], height[right]);
            ans = std::max(ans, sum);
            if(height[left] < height[right]) left ++;
            else right --;
        }
        return ans;
    }
};
```
