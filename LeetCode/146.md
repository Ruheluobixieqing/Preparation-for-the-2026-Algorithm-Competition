# 146.LRU 缓存

这题是我最近做过的知识点比较新的题目了，用到了 unordered_map，本质上是一个 hash table

个人感觉这种题目倒是有点接近于真实的应用了，值得好好回味一下

```cpp
struct DLinkedNode {
    int key, value;
    DLinkedNode* pre;
    DLinkedNode* next;
    // 默认构造函数
    DLinkedNode(): key(0), value(0), pre(nullptr), next(nullptr){}

    // 含参构造函数
    DLinkedNode(int _key, int _value):key(_key), value(_value), pre(nullptr), next(nullptr) {}
};

class LRUCache {
private:
    // unorderd_map 本质上是 hash table
    std::unordered_map<int, DLinkedNode*> cache;

    // 头尾指针
    DLinkedNode* head;
    DLinkedNode* tail;

    int size = 0;
    int capacity =0;

public:
    LRUCache(int _capacity):capacity(_capacity),size(0) {
        // 初始化 LRUcache
        head = new DLinkedNode();
        tail = new DLinkedNode();
        head -> next = tail;
        tail -> pre = head;
    }
    
    int get(int key) {
        // 没有找到，则返回 -1
        if(!cache.count(key))
        {
            return -1;
        }

        // 先定位结点位置，再转移至头部
        DLinkedNode* tar = cache[key];
        moveToHead(tar);
        return tar->value;
    }
    
    void put(int key, int value) {
        // 找到了，则更新其值
        if(cache.count(key))
        {
            DLinkedNode* tar = cache[key];
            tar -> value = value;
            moveToHead(tar);
            return;
        }

        // 没有找到,创建新结点
        DLinkedNode* newNode = new DLinkedNode(key, value);

        // 如果 capacity 已满
        if(size == capacity)
        {
            // 需要删除尾部结点腾出空间
            removeTail();
            size --;
        }
        cache[key] = newNode;
        addToHead(newNode);
        size++;
    }

    void removeNode(DLinkedNode* node)
    {
        node -> pre -> next = node -> next;
        node -> next -> pre = node -> pre;
    }

    void addToHead(DLinkedNode* node)
    {
        node -> pre = head;
        node -> next = head -> next;
        head -> next -> pre = node;
        head -> next = node;
    }

    void moveToHead(DLinkedNode* node)
    {
        removeNode(node);
        addToHead(node);
    }

    void removeTail(){
        DLinkedNode* lastNode = tail -> pre;
        removeNode(lastNode);

        // 从 cache 中删除记录
        cache.erase(lastNode->key);

        // 释放空间
        delete lastNode;
    }
};

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache* obj = new LRUCache(capacity);
 * int param_1 = obj->get(key);
 * obj->put(key,value);
 */
```
