# 22.括号生成

考虑到 n 较小，可以枚举答案

利用二进制字符串'0''1'分别表示'('')',枚举所有字符串，逐个检查是否合法即可

```cpp
class Solution {
public:
    bool check(std::string s){
        std::stack<char> st;
        for(const auto &e: s){
            if(e == '(') st.push(e);
            else {
                if(st.empty()) return false;
                else st.pop();
            }
        }
        return st.empty() ? true : false;
    }

    std::string intToString(int num, int n){
        std::string result;
        for(int i = n - 1; i>=0; i--){
            result += (num & (1 << i)) ? "(" : ")";
        }
        return result;
    }

    vector<string> generateParenthesis(int n) {
        std::vector<std::string> total;
        std::vector<std::string> ans;
        for(int i = 0; i <= std::pow(2, 2*n); i++)
        {
            std::string s = intToString(i, 2*n);
            if(check(s)) ans.push_back(s);
        }
        return ans;
    }
};
```
