# 54.螺旋矩阵

这题还是挺有意思的，一个看起来有些难但是实际上很好找到规律的模拟题

```cpp
class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int m = matrix[0].size();
        std::vector<int> ans;
        int sum = n * m;
        int x = 0, y = -1;
        while(sum)
        {
            for(int i = 1; i <= m && sum; i++)
            {
                y++;
                std::cout<<matrix[x][y]<< ' ';
                ans.push_back(matrix[x][y]);
                sum --;
            }
            m -= 1;
            for(int i = 1; i < n && sum; i++)
            {
                x ++;
                std::cout<<matrix[x][y]<< ' ';
                ans.push_back(matrix[x][y]);
                sum --;
            }
            n -= 1;
            for(int i = 1; i <= m && sum; i++)
            {
                y --;
                std::cout<<matrix[x][y]<< ' ';
                ans.push_back(matrix[x][y]);
                sum --;
            }
            m -= 1;
            for(int i = 1; i < n && sum; i++)
            {
                x --;
                std::cout<<matrix[x][y]<< ' ';
                ans.push_back(matrix[x][y]);
                sum --;
            }
            n -= 1;
        }
        return ans;
    }
};
```
