# 21.合并两个有序链表

直接在原链表上改动即可

```cpp
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        if(list1 == nullptr) return list2;
        if(list2 == nullptr) return list1;
        if(list1 -> val > list2 -> val) std::swap(list1, list2);
        ListNode* head = list1;
        list1 = list1 -> next;
        ListNode* p = head;
        while(list1 != nullptr || list2 != nullptr){
            if(list1 == nullptr){
                p -> next = list2;
                break;
            }
            if(list2 == nullptr){
                p -> next = list1;
                break;
            }
            if(list1 -> val <= list2 -> val)
            {
                p -> next = list1;
                p = p -> next;
                list1 = list1 -> next;
            }
            else {
                p -> next = list2;
                p = p -> next;
                list2 = list2 -> next;
            }
        }
        return head;
    }
};
```
