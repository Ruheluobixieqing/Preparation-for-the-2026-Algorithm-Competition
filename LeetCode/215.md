# 215.数组中第 K 个最大元素

大根堆

```cpp
class Solution {
public:
    std::priority_queue<int> pq;
    int findKthLargest(vector<int>& nums, int k) {
        for(const auto &e: nums) pq.push(e);
        while(--k) pq.pop();
        return pq.top();
    }
};
```

或者更简单，直接排序

```cpp
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        std::sort(nums.begin(), nums.end(), std::greater<int>());
        return nums[k-1];
    }
};
```
