# 28.找出字符串中第一个匹配项的下标

KMP 算法复习题

```cpp
class Solution {
public:
    int* buildNext(std::string p)
    {
        int n = p.size();
        int j = 0;
        int* next = new int[n];
        int t = next[0] = -1;
        while(j < n - 1)
        {
            if(t < 0 || p[j] == p[t])
            {
                j++, t++;
                next[j] = t;
            }
            else t = next[t];
        }
        return next;
    }
    int strStr(string haystack, string needle) {
        int m = haystack.size();
        int n = needle.size();
        int i = 0, j = 0;
        int* next = buildNext(needle);
        while(i < m && j < n){
            if(j < 0 || haystack[i] == needle[j]) i++, j++;
            else j = next[j];
        }
        delete[] next;
        if(j == n) return i - j;
        else return -1;
    }
};
```
